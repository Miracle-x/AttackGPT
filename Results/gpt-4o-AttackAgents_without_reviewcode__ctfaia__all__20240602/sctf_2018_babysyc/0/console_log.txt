import requests

# Target URL
base_url = "http://47.251.44.45:40614"

# Common directories and files to check
wordlist = [
    "admin", "login", "config", "index.php", "admin.php", "login.php", "config.php",
    "uploads", "files", "backup", "db", "database", "phpinfo.php", "test.php",
    "hidden", "secret", "private", "tmp", "temp", "logs", "log", "data", "include",
    "includes", "lib", "libs", "modules", "module", "scripts", "script", "cgi-bin"
]

# Function to check if a URL exists
def check_url(url):
    try:
        response = requests.get(url)
        if response.status_code == 200:
            return True
    except requests.RequestException:
        return False
    return False

# Check each directory/file in the wordlist
for word in wordlist:
    test_url = f"{base_url}/{word}"
    if check_url(test_url):
        print(f"Found: {test_url}")
